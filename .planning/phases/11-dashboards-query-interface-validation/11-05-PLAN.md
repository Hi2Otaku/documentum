```typescript
interface WorkflowQueryResponse {

  template_version: number;

  started_at: string | null;
  completed_at: string | null;
  active_activity: string | null;
}

interface WorkItemQueryResponse {
  id: string;
  activity_name: string;
  workflow_name: string;
  workflow_id: string;
  assignee: string | null;
  state: string;
  priority: number;
  created_at: string;
}

interface DocumentQueryResponse {
  id: string;
  title: string;
  lifecycle_state: string | null;
  current_version: string;
  author: string | null;
  created_by: string | null;
  updated_at: string;
  content_type: string;
}

// GET /api/v1/query/workflows?template_id=&state=&date_from=&date_to=&started_by=&skip=&limit=

// GET /api/v1/query/documents?lifecycle_state=&metadata_key=&metadata_value=&version=&skip=&limit=
// All return EnvelopeResponse<T[]> with PaginationMeta
```
