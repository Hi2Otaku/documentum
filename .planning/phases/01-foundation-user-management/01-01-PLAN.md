**src/app/models/base.py** (per D-07, D-08, D-10, D-06):
```python
import uuid
from datetime import datetime, timezone
from sqlalchemy import Boolean, DateTime, String, text
from sqlalchemy.dialects.postgresql import UUID
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column

class Base(DeclarativeBase):
    pass

class BaseModel(Base):
    __abstract__ = True
    id: Mapped[uuid.UUID] = mapped_column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4, server_default=text("gen_random_uuid()"))
    created_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), default=lambda: datetime.now(timezone.utc), server_default=text("now()"), nullable=False)
    updated_at: Mapped[datetime] = mapped_column(DateTime(timezone=True), default=lambda: datetime.now(timezone.utc), onupdate=lambda: datetime.now(timezone.utc), server_default=text("now()"), nullable=False)
    created_by: Mapped[str | None] = mapped_column(String(255), nullable=True)
    is_deleted: Mapped[bool] = mapped_column(Boolean, default=False, server_default=text("false"), nullable=False)
```

- class ActivityTemplate(BaseModel): __tablename__="activity_templates", process_template_id FK->process_templates.id, name String(255), activity_type Enum(ActivityType), description Text nullable, performer_type String(50) nullable, performer_id String(255) nullable, trigger_type String(20) default "or_join", position_x Float nullable, position_y Float nullable. - class FlowTemplate(BaseModel): __tablename__="flow_templates", process_template_id FK->process_templates.id, source_activity_id FK->activity_templates.id, target_activity_id FK->activity_templates.id, flow_type Enum(FlowType) default NORMAL, condition_expression Text nullable.

Runtime instance models:
- class WorkflowInstance(BaseModel): __tablename__="workflow_instances", process_template_id FK->process_templates.id, state Enum(WorkflowState) default DORMANT, started_at DateTime nullable, completed_at DateTime nullable, supervisor_id UUID nullable (FK->users.id).
- class ActivityInstance(BaseModel): __tablename__="activity_instances", workflow_instance_id FK->workflow_instances.id, activity_template_id FK->activity_templates.id, state String(50) default "dormant", started_at DateTime nullable, completed_at DateTime nullable.
- class WorkItem(BaseModel): __tablename__="work_items", activity_instance_id FK->activity_instances.id, performer_id UUID FK->users.id nullable, state Enum(WorkItemState) default AVAILABLE, instructions Text nullable, due_date DateTime nullable, priority int default 5, completed_at DateTime nullable.

Data models:
- class ProcessVariable(BaseModel): __tablename__="process_variables", process_template_id FK->process_templates.id nullable, workflow_instance_id FK->workflow_instances.id nullable, name String(255), variable_type String(20) (string/int/boolean/date), string_value Text nullable, int_value Integer nullable, bool_value Boolean nullable, date_value DateTime nullable.
