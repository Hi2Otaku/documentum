```python
class ProcessTemplateResponse(BaseModel):
    id: uuid.UUID
    name: str
    description: str | None
    version: int
    state: ProcessState  # "draft" | "active"
    is_installed: bool
    installed_at: datetime | None
    created_at: datetime
    updated_at: datetime
    created_by: str | None
    is_deleted: bool

class ActivityTemplateResponse(BaseModel):
    id: uuid.UUID
    process_template_id: uuid.UUID
    name: str
    activity_type: ActivityType  # "start" | "end" | "manual" | "auto"
    description: str | None
    performer_type: str | None  # "supervisor" | "user" | "group" | "sequential" | "runtime"
    performer_id: str | None
    trigger_type: TriggerType  # "and_join" | "or_join"
    method_name: str | None
    position_x: float | None
    position_y: float | None
    routing_type: str | None
    performer_list: list | None
    created_at: datetime
    updated_at: datetime
    is_deleted: bool

class FlowTemplateResponse(BaseModel):
    id: uuid.UUID
    process_template_id: uuid.UUID
    source_activity_id: uuid.UUID
    target_activity_id: uuid.UUID
    flow_type: FlowType  # "normal" | "reject"
    condition_expression: str | dict | None
    display_label: str | None
    created_at: datetime
    updated_at: datetime
    is_deleted: bool

class ProcessVariableResponse(BaseModel):
    id: uuid.UUID
    process_template_id: uuid.UUID | None
    workflow_instance_id: uuid.UUID | None
    name: str
    variable_type: str  # "string" | "int" | "boolean" | "date"
    string_value: str | None
    int_value: int | None
    bool_value: bool | None
    date_value: datetime | None

class ProcessTemplateDetailResponse(ProcessTemplateResponse):
    activities: list[ActivityTemplateResponse] = []
    flows: list[FlowTemplateResponse] = []
    variables: list[ProcessVariableResponse] = []

class ValidationErrorDetail(BaseModel):
    code: str
    message: str
    entity_type: str
    entity_id: str | None = None

class ValidationResult(BaseModel):
    valid: bool
    errors: list[ValidationErrorDetail] = []
```

       ```typescript
       export interface ActivityNodeData {
         name: string;
         description?: string;
         activityType: 'start' | 'end' | 'manual' | 'auto';
         performerType?: string | null;
         performerId?: string | null;
         triggerType?: 'and_join' | 'or_join';
         methodName?: string | null;
         routingType?: string | null;
         performerList?: string[] | null;
         // Tracks whether this node exists on server (has API id) or is new
         apiId?: string;
       }

       export interface FlowEdgeData {
         flowType: 'normal' | 'reject';
         conditionExpression?: string | null;
         displayLabel?: string | null;
         apiId?: string;
       }

       export type ActivityNodeType = 'startNode' | 'manualNode' | 'autoNode' | 'endNode';
       ```

    5. Create `frontend/src/api/client.ts` — Base API client:
       - Export `apiFetch<T>(path: string, options?: RequestInit): Promise<T>` function
       - Reads JWT token from `useAuthStore.getState().token`
       - Sets `Authorization: Bearer {token}` header if token exists
       - Sets `Content-Type: application/json` by default
       - On response: checks `res.ok`, throws `Error` with `body.detail || body.errors?.[0]?.message || res.statusText`
       - Unwraps `envelope.data as T` from the `EnvelopeResponse` wrapper
       - On 401: clear auth store (`useAuthStore.getState().logout()`), redirect to `/login` via `window.location.href = '/login'`
       - Export `API_BASE = '/api/v1'`

    6. Create `frontend/src/api/auth.ts`:
       - Export `loginApi(username: string, password: string): Promise<string>` (returns access_token)
       - Uses `fetch('/api/v1/auth/login', { method: 'POST', body: new URLSearchParams({ username, password }), headers: { 'Content-Type': 'application/x-www-form-urlencoded' } })`
       - CRITICAL: Must use URLSearchParams, NOT JSON.stringify — backend uses OAuth2PasswordRequestForm
       - Unwraps `envelope.data.access_token`

    11. Create `frontend/src/lib/serialization.ts` (per D-07, DESIGN-07):
        - `activitiesToNodes(activities: ActivityTemplateResponse[]): Node[]` — Maps each activity to a React Flow Node:
          - `id`: `String(activity.id)`
          - `type`: map activity_type 'start'->'startNode', 'manual'->'manualNode', 'auto'->'autoNode', 'end'->'endNode'
          - `position`: `{ x: activity.position_x ?? 0, y: activity.position_y ?? 0 }`
          - `data`: `{ name, description, activityType, performerType, performerId, triggerType, methodName, routingType, performerList, apiId: String(activity.id) }`
        - `flowsToEdges(flows: FlowTemplateResponse[]): Edge[]` — Maps each flow to a React Flow Edge:
          - `id`: `String(flow.id)`
          - `source`: `String(flow.source_activity_id)`
          - `target`: `String(flow.target_activity_id)`
          - `type`: if flow.condition_expression then 'conditionalEdge', else map flow_type 'normal'->'normalEdge', 'reject'->'rejectEdge'
          - `data`: `{ flowType, conditionExpression, displayLabel, apiId: String(flow.id) }`
        - `nodesToActivities(nodes: Node[]): ActivityTemplateCreate[]` — Reverse mapping for save (only data fields + position)
        - `edgesToFlows(edges: Edge[]): FlowTemplateCreate[]` — Reverse mapping for save
