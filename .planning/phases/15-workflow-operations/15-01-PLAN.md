```python
class WorkflowStartRequest(BaseModel):
    template_id: uuid.UUID
    document_ids: list[uuid.UUID] = []
    performer_overrides: dict[str, str] = {}
    initial_variables: dict[str, Any] = {}
    alias_set_id: uuid.UUID | None = None

class WorkflowInstanceResponse(BaseModel):
    id: uuid.UUID
    process_template_id: uuid.UUID
    state: WorkflowState  # dormant, running, halted, failed, finished
    started_at: datetime | None
    completed_at: datetime | None
    supervisor_id: uuid.UUID | None
    created_at: datetime
    updated_at: datetime

class WorkflowAdminListResponse(BaseModel):
    id: uuid.UUID
    process_template_id: uuid.UUID
    state: WorkflowState
    started_at: datetime | None
    completed_at: datetime | None
    supervisor_id: uuid.UUID | None
    template_name: str | None
    started_by_username: str | None
    active_activity_name: str | None
    created_at: datetime
    updated_at: datetime

class WorkflowDetailResponse(BaseModel):
    id: uuid.UUID
    process_template_id: uuid.UUID
    state: WorkflowState
    started_at: datetime | None
    completed_at: datetime | None
    supervisor_id: uuid.UUID | None
    activity_instances: list[ActivityInstanceResponse] = []
    work_items: list[WorkItemResponse] = []
    process_variables: list[ProcessVariableResponse] = []
    created_at: datetime
    updated_at: datetime

class ActivityInstanceResponse(BaseModel):
    id: uuid.UUID
    workflow_instance_id: uuid.UUID
    activity_template_id: uuid.UUID
    state: ActivityState  # dormant, active, paused, complete, error
    started_at: datetime | None
    completed_at: datetime | None
    created_at: datetime

class ProcessVariableResponse(BaseModel):
    id: uuid.UUID
    name: str
    variable_type: str
    string_value: str | None
    int_value: int | None
    bool_value: bool | None
    date_value: datetime | None

class WorkflowActionResponse(BaseModel):
    id: uuid.UUID
    state: WorkflowState
    message: str
```

Backend endpoints:
- POST /api/workflows — start workflow (body: WorkflowStartRequest)
- GET /api/workflows?skip=N&limit=N — list instances (returns EnvelopeResponse with data + meta)
- GET /api/workflows/admin/list?skip=N&limit=N&state=X&template_id=X&date_from=X&date_to=X — admin list
- GET /api/workflows/{id} — detail (returns EnvelopeResponse with WorkflowDetailResponse)
- POST /api/workflows/{id}/halt — halt (admin only)
- POST /api/workflows/{id}/resume — resume (admin only)
- POST /api/workflows/{id}/abort — terminate/abort (admin only)

All responses wrapped in EnvelopeResponse: { data: T, meta?: PaginationMeta }
PaginationMeta: { page, page_size, total_count, total_pages }
